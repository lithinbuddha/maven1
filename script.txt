pipeline { 
    agent any 
    tools { 
          maven 'MAVEN_HOME' //Ensure name matches with configured  
    } 
    stages { 
        stage('Checkout') {  
            steps { 
                git branch: 'master', url: 'https://github.com/lithinbuddha/maven1.git'  
            } 
    } 
     stage('Build') {  
            steps { 
                bat 'mvn clean package'  
            } 
      } 
     stage('Test') {  
            steps { 
                bat 'mvn test'  
            } 
      } 
     stage('Run Application') {  
            steps { 
                bat 'java -jar target/maven11-0.0.1-SNAPSHOT.jar'  
            } 
    }
    }
}
